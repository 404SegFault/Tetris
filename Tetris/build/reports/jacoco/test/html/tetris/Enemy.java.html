package demolition;
import demolition.*;
import processing.core.PImage;
import java.util.*;

public abstract class Enemy extends MovingSprite{
	public Enemy(PImage sprite, int x, int y, PImage[][] animations){
		super(sprite, x, y, animations);
		this.direction = Cardinal.DOWN;
	}

	/** 
	* The automatic AI movement of the enemies, called every second. will move according to which cardinal direction they are currently on
	* @param map so the enemy can check the collision objects
	* @return boolean whether the enemy hit the player or not
	**/
	public boolean movement(Gamemap map) {
		int[] newCoords = this.getCoords().clone();
		
		switch (this.direction){
			case UP:
				newCoords[1] -= App.GRIDSPACE;
				break;
			case DOWN:
				newCoords[1] += App.GRIDSPACE;
				break;
			case LEFT:
				newCoords[0] -= App.GRIDSPACE;
				break;
			case RIGHT:
				newCoords[0] += App.GRIDSPACE;
				break;
		}

		GameObject objectBeingCollided = map.findCollisionObject(this, newCoords); 

		if (objectBeingCollided instanceof Wall){
			this.collisionDecision(map);
			return false;
		}

		if (Arrays.equals(newCoords, map.getPlayer().getCoords())){
			return true;
		}

		newCoords[1] -= App.GRIDSPACE;
		this.setCoord(newCoords);
		
		return false;
	}

	/**When faced with a wall the enemies will make a decision on what direction they should go in next, this method is overriden by the class RedEnemy and Yellow Enemy.
	* The Red Enemy moves in a straight line and turns to a random decision every time its path is blocked by a solid/broken wall.
	* The Yellow Enemy moves in a straight line, but on collision with a wall will attempt to move clockwise
	* is called when faced with a wall, will continue to call movement until not blocked by a wall 
	**/
	public abstract void collisionDecision(Gamemap map);
}

class RedEnemy extends Enemy{
	private Random random = new Random();

	public RedEnemy(PImage sprite, int x, int y, PImage[][] animations){
		super(sprite, x, y, animations);
	}

	public void collisionDecision(Gamemap map){
		Cardinal randomPosition = Cardinal.values()[random.nextInt(4)];
		this.direction = randomPosition;
		movement(map);
	}
}

class YellowEnemy extends Enemy{
	private int currentIndex = 0;

	public YellowEnemy(PImage sprite, int x, int y, PImage[][] animations){
		super(sprite, x, y, animations);
	}

	public void collisionDecision(Gamemap map){
		this.direction = Cardinal.values()[currentIndex % 4];
		currentIndex++;
		movement(map);
	}
}
