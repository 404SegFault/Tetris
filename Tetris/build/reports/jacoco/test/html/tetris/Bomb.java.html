package demolition;
import demolition.*;
import processing.core.PApplet;
import processing.core.PImage;
import java.util.*;

public class Bomb extends GameObject {
	private int currentFrame; 
	private PImage[] animation; 
	private BombTimer bombTimer; 
	private Gamemap map; 

	public Bomb(PImage sprite, int x, int y, PImage[] animation, Gamemap map){
		super(sprite, x, y);
		this.animation = animation;
		this.map = map;

		Timer timer = new Timer();
		this.bombTimer = new BombTimer(this);
		//makes the bombTimer deduct 0.25 seconds every time
		timer.schedule(bombTimer, 0, 250);
	}

	/**moves to the next frame of the animation cycle**/
	public void nextFrame(){
		this.sprite = animation[currentFrame];
		currentFrame ++;
		currentFrame %= 8;
	}

	public Gamemap getMap() {return this.map;};
}
