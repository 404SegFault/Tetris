package demolition;
import demolition.*;

import processing.core.PApplet;
import processing.core.PImage;

import java.util.*;

public class Explosion extends GameObject {
	private HashMap<String, PImage> explosionSprites; 
	private HashMap<String, int[]> directionAndCoordinates; 
	private Gamemap map; 
	
	/**When an explosion object is created it also creates an ExplosionTimer which will remove the explosion after 0.5 seconds**/
	public Explosion(PImage sprite, int x, int y, HashMap<String, PImage> explosionSprites, HashMap<String, int[]> directionAndCoordinates, Gamemap map){
		super(sprite, x, y);
		this.explosionSprites = explosionSprites;
		this.directionAndCoordinates = directionAndCoordinates;
		this.map = map;

		Timer timer = new Timer();
		ExplosionTimer explosionTimer = new ExplosionTimer(this);
		timer.schedule(explosionTimer, 500, 500);
	}

	/**
	* Given a set of coordinates it creates a explosion  in all directions where the bomb's coordinates were, until those coordinates hit a wall 
	* where it will remove those coordinates so the sprite and explosion will not be drawn, if it hits a broken wall while being created the broken wall will be destroyed
	* @param map,coords needs the map to know where the breakable objects are and the coordinates of where the bomb should initially be placed
	**/
	public static void createExplosion(Gamemap map, int[] coords){
		// can check the coordinates around it and remove those from the list. then inside the draw loop through whats there?
		LinkedHashMap<String, PImage> explosionSprites = new LinkedHashMap<>(map.getAllExplosionSprites());
		LinkedHashMap<String, int[]> directionAndCoordinates = new LinkedHashMap<>(); //get the keyset from the explosions

		for (String key : explosionSprites.keySet()) {
			int[] newCoords = coords.clone(); 
			
			// depending on what the name of the String key is, it will adjust the future coordinates of the explosion
			switch (key){
				case("UP"):
					newCoords[1] -= 32;
					break;
				case("DOWN"):
					newCoords[1] += 32;
					break;
				case("RIGHT"):
					newCoords[0] += 32;
					break;
				case("LEFT"):
					newCoords[0] -= 32;
					break;
				case("UP_END"):
					newCoords[1] -= 64;
					break;
				case("DOWN_END"):
					newCoords[1] += 64;
					break;
				case("RIGHT_END"):
					newCoords[0] += 64;
					break;
				case("LEFT_END"):
					newCoords[0] -= 64;
					break;
			}

			directionAndCoordinates.put(key, newCoords);
		}

		// goes through all the directions the explosion will end up 
		for (String direction: explosionSprites.keySet()){
			if (directionAndCoordinates.get(direction) != null){
				
				// gets the coordinates associated with that dirction, if the bomb was at 0,0 then DOWN_END will be [0,64]
				int[] coordinates = directionAndCoordinates.get(direction).clone();
				GameObject objectBeingCollided = map.findCollisionObject(coordinates);
				
				if (objectBeingCollided != null){
					// if the coordinate is a wall the explosion should be cut off
					if (objectBeingCollided.getClass().getSimpleName().equals("Wall")){
						
						//if a coordinate was found like left, then it should remove the explosion saying left
						directionAndCoordinates.remove(direction);
						//it should also remove the end of left
						directionAndCoordinates.remove(direction + "_END");
					}
					
					if (objectBeingCollided.getClass().getSimpleName().equals("BrokenWall")){
						objectBeingCollided.removeObject();
						directionAndCoordinates.remove(direction + "_END");
					}
				}
			}
		}

		Explosion newExplosion = new Explosion(explosionSprites.get("centre"), coords[0], coords[1], explosionSprites, directionAndCoordinates, map);
		map.addNewObject(newExplosion);
	}

	//**Checks all of the coordinates within its hashmap and destroys movingSprites that enter it**/
	public void tick(){
		// goes through all of the directions within the object (because of the createExplosion method it will be missing some directions)
		for (String direction : directionAndCoordinates.keySet()){
			
			int[] coordinates = directionAndCoordinates.get(direction);
			GameObject collisionObject = map.findCollisionObject(this, coordinates);
			// checks if there are any sprites walking on it then calls the appropriate methods
			if (collisionObject != null){
				if (collisionObject instanceof Enemy){
					collisionObject.removeObject();
				}

				if (collisionObject instanceof Player){
					map.playerHit();
				}
			}
		}
	}

	/**
	*calls tick and draws all the sprites that correspond to the hashmap
	* @param app the app where the explosion will be drawn
	**/
	public void draw(App app){
		tick();

		for (String key: directionAndCoordinates.keySet()){
			PImage sprite = explosionSprites.get(key);
			int[] explosionCoord = directionAndCoordinates.get(key);
			
			app.image(sprite, explosionCoord[0], explosionCoord[1]);
		}
	}
}
