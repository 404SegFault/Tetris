package demolition;
import demolition.*;
import processing.core.PImage;
import java.util.*;

public abstract class MovingSprite extends GameObject{
	protected Cardinal direction;
	protected PImage[][] allAnimations;
	protected int currentFrame = 0;
	
	/**when a sprite is created an animation timer is created to move the frames to the next in the cycle**/
	public MovingSprite(PImage sprite, int x, int y, PImage[][] allAnimations){
		// needs to be offset because the y of a moving sprite is higher 
		super(sprite, x, y - App.GRIDSPACE);
		this.allAnimations = allAnimations;

		Timer timer = new Timer();
		AnimationTimer animationTimer = new AnimationTimer(this);
		timer.schedule(animationTimer, 0, 250);
	}

	/**uses their direction to choose cycle is currently being used and increments the current frame the object is on**/
	public void nextFrame(){
		PImage[] currentCycle = null;

		switch (this.direction){
			case UP:
				currentCycle = this.allAnimations[0];
				break;
			case RIGHT:
				currentCycle = this.allAnimations[1];
				break;
			case DOWN:
				currentCycle = this.allAnimations[2];
				break;
			case LEFT:
				currentCycle = this.allAnimations[3];
				break;
		}

		this.sprite = currentCycle[currentFrame];
		currentFrame++;
		currentFrame %= 4;
	}

	/////////////////////////////////////////GETTERS/////////////////////////////////////////
	public Cardinal getDirection() {return this.direction;}
	/**Override the getCoords inside GameObject, because the sprites are bigger whenever coordinates are retrieved it to be at a lower position to get the correct collision coordinates**/
	public int[] getCoords(){
		int[] collisionCoords = this.coords.clone();
		collisionCoords[1] += App.GRIDSPACE;
		return collisionCoords;
	}
}

enum Cardinal {
	UP,
	RIGHT,
	DOWN,
	LEFT;
}
